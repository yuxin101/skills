import pandas as pd import numpy as np import yfinance as yf import matplotlib.pyplot as plt import warnings warnings.filterwarnings('ignore') # ===================== Basic Settings ===================== # plt.rcParams['font.sans-serif'] = ['SimHei'] # Adapt to Windows system # plt.rcParams['font.sans-serif'] = ['Arial Unicode MS'] # Adapt to Mac system plt.rcParams['axes.unicode_minus'] = False # Display minus signs normally # ===================== Data Acquisition ===================== # Get daily price data for TQQQ over the past 10 years (downloaded from Yahoo Finance) end_date = pd.Timestamp.today() start_date = end_date - pd.Timedelta(days=10*365) # Past 10 years ticker = 'TQQQ' # TQQQ 3x Long NASDAQ ETF code # Download data and preprocess df = yf.download(ticker, start=start_date, end=end_date) df = df.rename(columns={'Close': 'close', 'Open': 'open', 'High': 'high', 'Low': 'low', 'Volume': 'volume'}) df.reset_index(inplace=True) df.rename(columns={'Date': 'date'}, inplace=True) # ===================== Moving Average and Signal Calculation ===================== # Calculate 20-day/80-day moving average df['SMA20'] = df['close'].rolling(window=20).mean() df['SMA80'] = df['close'].rolling(window=80).mean() # Remove leading missing values caused by moving average calculation df = df.dropna(subset=['SMA20', 'SMA80']).reset_index(drop=True) # Define crossover/crossunder conditions (Core strategy logic) # Crossover: Today's closing price > max of 20/80-day MA, and did not crossover the previous day → Increase Holdings (Full position) # Crossunder: Today's closing price < min of 20/80-day MA, and did not crossunder the previous day → Sell All (Empty position) df['above_ma'] = df['close'] > df[['SMA20', 'SMA80']].max(axis=1) # Crossover max of any MA df['below_ma'] = df['close'] < df[['SMA20', 'SMA80']].min(axis=1) # Crossunder min of any MA # Previous day's status (Avoid look-ahead bias) df['prev_above_ma'] = df['above_ma'].shift(1) df['prev_below_ma'] = df['below_ma'].shift(1) # Generate buy/sell signals (Triggered only when status changes) df['buy_signal'] = (df['above_ma'] == True) & (df['prev_above_ma'] == False) # Buy signal df['sell_signal'] = (df['below_ma'] == True) & (df['prev_below_ma'] == False) # Sell signal # Generate position signals (0 = empty position, 1 = full position) df['position'] = 0 # Initial empty position df.loc[df['buy_signal'], 'position'] = 1 # Buy signal → full position df.loc[df['sell_signal'], 'position'] = 0 # Sell signal → empty position df['position'] = df['position'].ffill() # Forward fill position status (Hold position until sold) # Crucial: Shift position signal by 1 day (Avoid look-ahead bias, today's signal executed the next day) df['position'] = df['position'].shift(1).fillna(0) # ===================== Return and Drawdown Calculation ===================== # Calculate daily return df['asset_return'] = df['close'].pct_change() # Daily return of the target itself df['strategy_return'] = df['position'] * df['asset_return'] # Daily return of the strategy # Calculate cumulative return (Initial principal = 1) df['asset_cum_return'] = (1 + df['asset_return']).cumprod() df['strategy_cum_return'] = (1 + df['strategy_return']).cumprod() # Define drawdown calculation function def calculate_drawdown(cum_return_series): """Calculate drawdown series: (Current cumulative return / Historical maximum) - 1""" running_max = cum_return_series.cummax() drawdown = (cum_return_series / running_max) - 1 return drawdown # Calculate drawdown df['asset_drawdown'] = calculate_drawdown(df['asset_cum_return']) df['strategy_drawdown'] = calculate_drawdown(df['strategy_cum_return']) # Core metrics calculation total_strategy_return = (df['strategy_cum_return'].iloc[-1] - 1) * 100 # Total strategy return total_asset_return = (df['asset_cum_return'].iloc[-1] - 1) * 100 # Total target return max_strategy_drawdown = df['strategy_drawdown'].min() * 100 # Strategy maximum drawdown max_asset_drawdown = df['asset_drawdown'].min() * 100 # Target maximum drawdown avg_strategy_drawdown = df['strategy_drawdown'].mean() * 100 # Strategy average drawdown avg_asset_drawdown = df['asset_drawdown'].mean() * 100 # Target average drawdown # ===================== Visualization ===================== fig, (ax1, ax2) = plt.subplots(2, 1, figsize=(14, 10), sharex=True) # Subplot 1: Price trend + Moving Averages + Buy/Sell points ax1.plot(df['date'], df['close'], label='TQQQ Closing Price', color='blue', linewidth=1) ax1.plot(df['date'], df['SMA20'], label='20-Day MA', color='orange', linewidth=1) ax1.plot(df['date'], df['SMA80'], label='80-Day MA', color='red', linewidth=1) # Mark buy points (Green up arrows) ax1.scatter(df[df['buy_signal']]['date'], df[df['buy_signal']]['close'], marker='^', color='green', s=100, label='Buy Point', zorder=5) # Mark sell points (Red down arrows) ax1.scatter(df[df['sell_signal']]['date'], df[df['sell_signal']]['close'], marker='v', color='red', s=100, label='Sell Point', zorder=5) ax1.set_title('TQQQ Price Trend + 20/80-Day MA + Buy/Sell Points', fontsize=14) ax1.set_ylabel('Price (USD)', fontsize=12) ax1.legend(loc='upper left') ax1.grid(True, alpha=0.3) # Subplot 2: Strategy vs Target cumulative return ax2.plot(df['date'], df['asset_cum_return'], label='TQQQ Itself Cumulative Return', color='blue', linewidth=1) ax2.plot(df['date'], df['strategy_cum_return'], label='MA Strategy Cumulative Return', color='purple', linewidth=1) ax2.set_title('TQQQ Target vs MA Strategy Cumulative Return', fontsize=14) ax2.set_xlabel('Date', fontsize=12) ax2.set_ylabel('Cumulative Return (Initial Principal = 1)', fontsize=12) ax2.legend(loc='upper left') ax2.grid(True, alpha=0.3) plt.tight_layout() plt.show() # ===================== Output Results ===================== print("="*60) print("TQQQ 3x Long NASDAQ ETF MA Strategy Backtest Results (Past 10 Years)") print("="*60) print(f"Total Strategy Return: {total_strategy_return:.2f}%") print(f"Total Target Return: {total_asset_return:.2f}%") print(f"Strategy Max Drawdown: {max_strategy_drawdown:.2f}%") print(f"Target Max Drawdown: {max_asset_drawdown:.2f}%") print(f"Strategy Avg Drawdown: {avg_strategy_drawdown:.2f}%") print(f"Target Avg Drawdown: {avg_asset_drawdown:.2f}%") print("="*60)
