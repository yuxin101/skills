import pandas as pd import numpy as np import yfinance as yf import matplotlib.pyplot as plt import warnings warnings.filterwarnings('ignore') # ===================== Basic Settings ===================== # plt.rcParams['font.sans-serif'] = ['SimHei'] # Adapt to Windows system # plt.rcParams['font.sans-serif'] = ['Arial Unicode MS'] # Adapt to Mac system plt.rcParams['axes.unicode_minus'] = False # Display minus signs normally # ===================== Data Acquisition ===================== # Get daily price data for TQQQ over the past 10 years (downloaded from Yahoo Finance) end_date = pd.Timestamp.today() start_date = end_date - pd.Timedelta(days=10*365) # Past 10 years ticker = 'TQQQ' # TQQQ 3x Long NASDAQ ETF code # Download data and preprocess df = yf.download(ticker, start=start_date, end=end_date) df = df.rename(columns={'Close': 'close', 'Open': 'open', 'High': 'high', 'Low': 'low', 'Volume': 'volume'}) df.reset_index(inplace=True) df.rename(columns={'Date': 'date'}, inplace=True) # ===================== Moving Average and Signal Calculation ===================== # Calculate 20-day/80-day moving average df['SMA20'] = df['close'].rolling(window=20).mean() df['SMA80'] = df['close'].rolling(window=80).mean() # Remove leading missing values caused by moving average calculation df = df.dropna(subset=['SMA20', 'SMA80']).reset_index(drop=True) # Define crossover/crossunder conditions (Core strategy logic) # Crossover: Today's closing price > max of 20/80-day MA, and did not crossover the previous day → Increase Holdings (Full position) # Crossunder: Today's closing price < min of 20/80-day MA, and did not crossunder the previous day → Sell All (Empty position) df['above_ma'] = df['close'] > df[['SMA20', 'SMA80']].max(axis=1) # Crossover max of any MA df['below_ma'] = df['close'] < df[['SMA20', 'SMA80']].min(axis=1) # Crossunder min of any MA # Previous day's status (Avoid look-ahead bias) df['prev_above_ma'] = df['above_ma'].shift(1) df['prev_below_ma'] = df['below_ma'].shift(1) # Generate buy/sell signals (Triggered only when status changes) df['buy_signal'] = (df['above_ma'] == True) & (df['prev_above_ma'] == False) # Buy signal df['sell_signal'] = (df['below_ma'] == True) & (df['prev_below_ma'] == False) # Sell signal # Generate position signals (0 = empty position, 1 = full position) df['position'] = 0 # Initial empty position df.loc[df['buy_signal'], 'position'] = 1 # Buy signal → full position df.loc[df['sell_signal'], 'position'] = 0 # Sell signal → empty position df['position'] = df['position'].ffill() # Forward fill position status (Hold position until sold) # Crucial: Shift position signal by 1 day (Avoid look-ahead bias, today's signal executed the next day) df['position'] = df['position'].shift(1).fillna(0) # ===================== Return and Drawdown Calculation ===================== # Calculate daily return df['asset_return'] = df['close'].pct_change() # Daily return of the target itself df['strategy_return'] = df['position'] * df['asset_return'] # Daily return of the strategy # Calculate cumulative return (Initial principal = 1) df['asset_cum_return'] = (1 + df['asset_return']).cumprod() df['strategy_cum_return'] = (1 + df['strategy_return']).cumprod() # Define drawdown calculation function def calculate_drawdown(cum_return_series): """Calculate drawdown series: (Current cumulative return / Historical maximum) - 1""" running_max = cum_return_series.cummax() drawdown = (cum_return_series / running_max) - 1 return drawdown # Calculate drawdown df['asset_drawdown'] = calculate_drawdown(df['asset_cum_return']) df['strategy_drawdown'] = calculate_drawdown(df['strategy_cum_return']) # Core metrics calculation total_strategy_return = (df['strategy_cum_return'].iloc[-1] - 1) * 100 # Total strategy return total_asset_return = (df['asset_cum_return'].iloc[-1] - 1) * 100 # Total target return max_strategy_drawdown = df['strategy_drawdown'].min() * 100 # Strategy maximum drawdown max_asset_drawdown = df['asset_drawdown'].min() * 100 # Target maximum drawdown avg_strategy_drawdown = df['strategy_drawdown'].mean() * 100 # Strategy average drawdown avg_asset_drawdown = df['asset_drawdown'].mean() * 100 # Target average drawdown # ===================== Visualization ===================== fig, (ax1, ax2) = plt.subplots(2, 1, figsize=(14, 10), sharex=True) # Subplot 1: Price trend + Moving Averages + Buy/Sell points ax1.plot(df['date'], df['close'], label='TQQQ Closing Price', color='blue', linewidth=1) ax1.plot(df['date'], df['SMA20'], label='20-Day MA', color='orange', linewidth=1) ax1.plot(df['date'], df['SMA80'], label='80-Day MA', color='red', linewidth=1) # Mark buy points (Green up arrows) ax1.scatter(df[df['buy_signal']]['date'], df[df['buy_signal']]['close'], marker='^', color='green', s=100, label='Buy Point', zorder=5) # Mark sell points (Red down arrows) ax1.scatter(df[df['sell_signal']]['date'], df[df['sell_signal']]['close'], marker='v', color='red', s=100, label='Sell Point', zorder=5) ax1.set_title('TQQQ Price Trend + 20/80-Day MA + Buy/Sell Points', fontsize=14) ax1.set_ylabel('Price (USD)', fontsize=12) ax1.legend(loc='upper left') ax1.grid(True, alpha=0.3) # Subplot 2: Strategy vs Target cumulative return ax2.plot(df['date'], df['asset_cum_return'], label='TQQQ Itself Cumulative Return', color='blue', linewidth=1) ax2.plot(df['date'], df['strategy_cum_return'], label='MA Strategy Cumulative Return', color='purple', linewidth=1) ax2.set_title('TQQQ Target vs MA Strategy Cumulative Return', fontsize=14) ax2.set_xlabel('Date', fontsize=12) ax2.set_ylabel('Cumulative Return (Initial Principal = 1)', fontsize=12) ax2.legend(loc='upper left') ax2.grid(True, alpha=0.3) plt.tight_layout() plt.show() # pd.Timedelta(days=10*365) # 近10年 ticker = 'TQQQ' # TQQQ三倍做多纳指ETF代码 # 下载数据并预处理 df = yf.download(ticker, start=start_date, end=end_date) df = df.rename(columns={'Close': 'close', 'Open': 'open', 'High': 'high', 'Low': 'low', 'Volume': 'volume'}) df.reset_index(inplace=True) df.rename(columns={'Date': 'date'}, inplace=True) # ===================== 均线与信号计算 ===================== # 计算20日/80日均线 df['SMA20'] = df['close'].rolling(window=20).mean() df['SMA80'] = df['close'].rolling(window=80).mean() # 去除均线计算导致的前导缺失值 df = df.dropna(subset=['SMA20', 'SMA80']).reset_index(drop=True) # 定义上穿/下穿条件（核心策略逻辑） # 上穿：当日收盘价 > 20/80日均线最大值，且前一日未上穿 → 增持（满仓） # 下穿：当日收盘价 < 20/80日均线最小值，且前一日未下穿 → 全部卖出（空仓） df['above_ma'] = df['close'] > df[['SMA20', 'SMA80']].max(axis=1) # 上穿任意均线最大值 df['below_ma'] = df['close'] < df[['SMA20', 'SMA80']].min(axis=1) # 下穿任意均线最小值 # 前一日的状态（避免未来函数） df['prev_above_ma'] = df['above_ma'].shift(1) df['prev_below_ma'] = df['below_ma'].shift(1) # 生成买卖信号（仅在状态变化时触发） df['buy_signal'] = (df['above_ma'] == True) & (df['prev_above_ma'] == False) # 买入信号 df['sell_signal'] = (df['below_ma'] == True) & (df['prev_below_ma'] == False) # 卖出信号 # 生成持仓信号（0=空仓，1=满仓） df['position'] = 0 # 初始空仓 df.loc[df['buy_signal'], 'position'] = 1 # 买入信号→满仓 df.loc[df['sell_signal'], 'position'] = 0 # 卖出信号→空仓 df['position'] = df['position'].ffill() # 填充持仓状态（保持持仓直到卖出） # 关键：持仓信号移位1天（避免未来函数，当天信号次日执行） df['position'] = df['position'].shift(1).fillna(0) # ===================== 收益与回撤计算 ===================== # 计算日收益率 df['asset_return'] = df['close'].pct_change() # 标的本身日收益 df['strategy_return'] = df['position'] * df['asset_return'] # 策略日收益 # 计算累计收益率（初始本金=1） df['asset_cum_return'] = (1 + df['asset_return']).cumprod() df['strategy_cum_return'] = (1 + df['strategy_return']).cumprod() # 定义回撤计算函数 def calculate_drawdown(cum_return_series): """计算回撤序列：(当前累计收益/历史最大值) - 1""" running_max = cum_return_series.cummax() drawdown = (cum_return_series / running_max) - 1 return drawdown # 计算回撤 df['asset_drawdown'] = calculate_drawdown(df['asset_cum_return']) df['strategy_drawdown'] = calculate_drawdown(df['strategy_cum_return']) # 核心指标计算 total_strategy_return = (df['strategy_cum_return'].iloc[-1] - 1) * 100 # 策略总收益率 total_asset_return = (df['asset_cum_return'].iloc[-1] - 1) * 100 # 标的总收益率 max_strategy_drawdown = df['strategy_drawdown'].min() * 100 # 策略最大回撤 max_asset_drawdown = df['asset_drawdown'].min() * 100 # 标的最大回撤 avg_strategy_drawdown = df['strategy_drawdown'].mean() * 100 # 策略平均回撤 avg_asset_drawdown = df['asset_drawdown'].mean() * 100 # 标的平均回撤 # ===================== 可视化 ===================== fig, (ax1, ax2) = plt.subplots(2, 1, figsize=(14, 10), sharex=True) # 子图1：价格走势 + 均线 + 买卖点 ax1.plot(df['date'], df['close'], label='TQQQ收盘价', color='blue', linewidth=1) ax1.plot(df['date'], df['SMA20'], label='20日均线', color='orange', linewidth=1) ax1.plot(df['date'], df['SMA80'], label='80日均线', color='red', linewidth=1) # 标注买入点（绿色上箭头） ax1.scatter(df[df['buy_signal']]['date'], df[df['buy_signal']]['close'], marker='^', color='green', s=100, label='买入点', zorder=5) # 标注卖出点（红色下箭头） ax1.scatter(df[df['sell_signal']]['date'], df[df['sell_signal']]['close'], marker='v', color='red', s=100, label='卖出点', zorder=5) ax1.set_title('TQQQ 价格走势 + 20/80日均线 + 买卖点', fontsize=14) ax1.set_ylabel('价格（美元）', fontsize=12) ax1.legend(loc='upper left') ax1.grid(True, alpha=0.3) # 子图2：策略vs标的累计收益 ax2.plot(df['date'], df['asset_cum_return'], label='TQQQ本身累计收益', color='blue', linewidth=1) ax2.plot(df['date'], df['strategy_cum_return'], label='均线策略累计收益', color='purple', linewidth=1) ax2.set_title('TQQQ 标的 vs 均线策略 累计收益', fontsize=14) ax2.set_xlabel('日期', fontsize=12) ax2.set_ylabel('累计收益（初始本金=1）', fontsize=12) ax2.legend(loc='upper left') ax2.grid(True, alpha=0.3) plt.tight_layout() plt.show() # ===================== 结果输出 ===================== print("="*60) print("TQQQ 三倍做多纳指ETF 均线策略回测结果（近10年）") print("="*60) print(f"策略总收益率: {total_strategy_return:.2f}%") print(f"标的总收益率: {total_asset_return:.2f}%") print(f"策略最大回撤: {max_strategy_drawdown:.2f}%") print(f"标的最大回撤: {max_asset_drawdown:.2f}%") print(f"策略平均回撤: {avg_strategy_drawdown:.2f}%") print(f"标的平均回撤: {avg_asset_drawdown:.2f}%") print("="*60)

# AiroomLtd Global Finance Data Platform

An OpenClaw skill for downloading financial data from the airoom.ltd WordPress site.

## Purpose

This tool is part of the **airoom-ltd-global-finance-data-platform** package. It is designed specifically to download financial data files from the airoom.ltd WordPress site.

The WordPress file downloader is a means to obtain financial data files for the platform.

## IMPORTANT: No Login Required

**For http://airoom.ltd/index.php/airoom/ - No login is required.**

This page is publicly accessible. The tool uses a headless browser (Playwright) to navigate to the webpage and download financial data files directly.

## Installation

```bash
# Install Python dependencies
pip install -r requirements.txt

# Install Playwright browser (Chromium)
playwright install chromium
```

## Configuration

### Environment Variables

```bash
export WP_URL="http://airoom.ltd"
export WP_TARGET_URL="http://airoom.ltd/index.php/airoom/"
export WP_OUTPUT_DIR="./downloads"
export WP_MAX_FILES="0"
```

Or create config file at `~/.config/airoom-ltd-global-finance-data-platform/config.json`:

```json
{
  "wordpress": {
    "url": "http://airoom.ltd"
  },
  "target": {
    "page_url": "http://airoom.ltd/index.php/airoom/"
  },
  "download": {
    "output_dir": "./downloads",
    "max_files": 0
  }
}
```

## Usage

```bash
# Download financial data files
python3 main.py download

# Test connection
python3 main.py test

# Show configuration
python3 main.py config
```

## How It Works

1. **Headless Browser**: Uses Playwright (headless Chromium) to navigate web pages
2. **Page Access**: Connects to the target WordPress page (no login needed for public pages)
3. **File Detection**: Scans the page for downloadable financial data files
4. **Download**: Saves financial data files to the local output directory

This is standard web automation - the browser makes network requests to load web pages and extract financial data.
