    const opacityContents = scrollYProgress.map(progress =>

        useTransform(progress, [0, 0.7], ["inset(0 100% 0 0)", "inset(0 0% 0 0)"])
    );

        useTransform(progress, [0, 1], [-50, 0])

  return (
